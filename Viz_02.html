pandas Scatter Plot¶

In [7]:
import matplotlib.pyplot as plt
import pandas as pd

temperature = [25.2, 27.4, 22.9, 26.2, 29.5, 33.1, 30.4, 36.1, 34.4, 29.1]
Ice_cream_sales = [236500,357500, 203500, 365200, 446600, 574200, 453200, 675400,598400, 463100]

dict_data = {'t':temperature, 'sales_amt':Ice_cream_sales}
df_ice_cream = pd.DataFrame(dict_data, columns=['t','sales_amt'])

print(df_ice_cream)

df_ice_cream.plot.scatter(x='t', y='sales_amt', grid=True, title='temp & sales_amt')
plt.show()

pandas Bar Plot¶

In [8]:
import matplotlib.pyplot as plt
import pandas as pd
%matplotlib inline

grade_num = [5, 14, 12, 3]
students = ['A', 'B', 'C', 'D']

df_grade = pd.DataFrame(grade_num, index=students, columns=['Student'])
print(df_grade)

grade_bar = df_grade.plot.bar(grid=True)
grade_bar.set_xlabel("grade")
grade_bar.set_ylabel("number of students")
grade_bar.set_title("bar plot of number of students per grades")
